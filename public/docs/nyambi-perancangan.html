1. Gambaran Umum Sistem

Arsitektur dan scope platform Nyambi

Nyambi adalah platform marketplace yang menghubungkan pengguna dengan pekerja terampil harian di Indonesia — tukang kebun, listrik, cleaning service, bangunan, dan 45+ kategori lainnya.

2.3 DDL Lengkap (SQL)

-- ============================================================
-- NYAMBI DATABASE SCHEMA v1.0
-- PostgreSQL 15+
-- ============================================================

CREATE EXTENSION IF NOT EXISTS "pgcrypto";

-- ── 1. USERS ──────────────────────────────────────────────
CREATE TABLE users (
  id           UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
  nama         VARCHAR(100) NOT NULL,
  email        VARCHAR(150) UNIQUE NOT NULL,
  telepon      VARCHAR(20),
  password     VARCHAR(255) NOT NULL,      -- bcrypt hash
  is_worker    BOOLEAN      DEFAULT FALSE,
  created_at   TIMESTAMP    DEFAULT NOW(),
  updated_at   TIMESTAMP    DEFAULT NOW()
);

-- ── 2. CATEGORIES ─────────────────────────────────────────
CREATE TABLE categories (
  id         VARCHAR(50)  PRIMARY KEY,     -- e.g. 'kebun', 'listrik'
  icon       VARCHAR(50)  NOT NULL,        -- Material Symbol name
  title      VARCHAR(100) NOT NULL,
  created_at TIMESTAMP    DEFAULT NOW()
);

-- ── 3. WORKER_PROFILES ────────────────────────────────────
CREATE TABLE worker_profiles (
  id               UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id          UUID         UNIQUE REFERENCES users(id) ON DELETE CASCADE,
  specialty        VARCHAR(100) NOT NULL,
  bio              TEXT,
  location         VARCHAR(100),
  status           VARCHAR(20)  DEFAULT 'available'
                     CHECK (status IN ('available', 'busy')),
  experience_years SMALLINT     DEFAULT 0,
  response_time    VARCHAR(50),
  completed_jobs   INT          DEFAULT 0,
  rating           NUMERIC(3,2) DEFAULT 0.00,
  image_url        TEXT,
  created_at       TIMESTAMP    DEFAULT NOW(),
  updated_at       TIMESTAMP    DEFAULT NOW()
);

-- ── 4. WORKER_CATEGORIES ──────────────────────────────────
CREATE TABLE worker_categories (
  worker_id   UUID        REFERENCES worker_profiles(id) ON DELETE CASCADE,
  category_id VARCHAR(50) REFERENCES categories(id)     ON DELETE CASCADE,
  PRIMARY KEY (worker_id, category_id)
);

-- ── 5. WORKER_TAGS ────────────────────────────────────────
CREATE TABLE worker_tags (
  id        SERIAL      PRIMARY KEY,
  worker_id UUID        REFERENCES worker_profiles(id) ON DELETE CASCADE,
  tag       VARCHAR(50) NOT NULL
);

-- ── 6. WORKER_SERVICES ────────────────────────────────────
CREATE TABLE worker_services (
  id        SERIAL       PRIMARY KEY,
  worker_id UUID         REFERENCES worker_profiles(id) ON DELETE CASCADE,
  name      VARCHAR(100) NOT NULL,
  price     INT,                           -- harga dalam Rupiah
  unit      VARCHAR(30)                    -- 'per kunjungan', 'per jam'
);

-- ── 7. ORDERS ─────────────────────────────────────────────
CREATE TABLE orders (
  id         UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    UUID        REFERENCES users(id)           ON DELETE SET NULL,
  worker_id  UUID        REFERENCES worker_profiles(id) ON DELETE SET NULL,
  tanggal    DATE        NOT NULL,
  waktu      TIME        NOT NULL,
  deskripsi  TEXT,
  alamat     TEXT        NOT NULL,
  telepon    VARCHAR(20) NOT NULL,
  status     VARCHAR(20) DEFAULT 'menunggu'
               CHECK (status IN ('menunggu','dikonfirmasi','selesai','dibatalkan')),
  created_at TIMESTAMP   DEFAULT NOW(),
  updated_at TIMESTAMP   DEFAULT NOW()
);

-- ── 8. REVIEWS ────────────────────────────────────────────
CREATE TABLE reviews (
  id         UUID     PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id   UUID     UNIQUE REFERENCES orders(id)           ON DELETE CASCADE,
  user_id    UUID     REFERENCES users(id)                   ON DELETE SET NULL,
  worker_id  UUID     REFERENCES worker_profiles(id)         ON DELETE CASCADE,
  rating     SMALLINT NOT NULL CHECK (rating BETWEEN 1 AND 5),
  comment    TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- ── 9. FAVORITES ──────────────────────────────────────────
CREATE TABLE favorites (
  user_id    UUID      REFERENCES users(id)           ON DELETE CASCADE,
  worker_id  UUID      REFERENCES worker_profiles(id) ON DELETE CASCADE,
  created_at TIMESTAMP DEFAULT NOW(),
  PRIMARY KEY (user_id, worker_id)
);

-- ── 10. NOTIFICATIONS ─────────────────────────────────────
CREATE TABLE notifications (
  id         UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    UUID         REFERENCES users(id) ON DELETE CASCADE,
  icon       VARCHAR(50)  NOT NULL,
  title      VARCHAR(150) NOT NULL,
  body       TEXT,
  href       VARCHAR(255),
  is_read    BOOLEAN      DEFAULT FALSE,
  created_at TIMESTAMP    DEFAULT NOW()
);

2.4 Index & Trigger

-- ── INDEX untuk performa query ────────────────────────────
CREATE INDEX idx_workers_location  ON worker_profiles(location);
CREATE INDEX idx_workers_status    ON worker_profiles(status);
CREATE INDEX idx_workers_rating    ON worker_profiles(rating DESC);
CREATE INDEX idx_orders_user_id    ON orders(user_id);
CREATE INDEX idx_orders_worker_id  ON orders(worker_id);
CREATE INDEX idx_orders_status     ON orders(status);
CREATE INDEX idx_reviews_worker_id ON reviews(worker_id);
CREATE INDEX idx_notifs_user_read  ON notifications(user_id, is_read);

-- ── TRIGGER: auto-update rating setelah review baru ───────
CREATE OR REPLACE FUNCTION update_worker_rating()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE worker_profiles
  SET rating = (
    SELECT ROUND(AVG(rating)::NUMERIC, 2)
    FROM reviews
    WHERE worker_id = NEW.worker_id
  )
  WHERE id = NEW.worker_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_rating
AFTER INSERT OR UPDATE ON reviews
FOR EACH ROW EXECUTE FUNCTION update_worker_rating();

5. Catatan Implementasi

Hal-hal penting saat membangun backend

Aturan Bisnis Penting

FiturAturan
ReviewHanya bisa ditulis jika status order = selesai dan belum ada review untuk order tersebut (UNIQUE constraint pada order_id)
Batalkan OrderHanya bisa jika status = menunggu. Setelah dikonfirmasi tidak bisa dibatalkan via API biasa
Daftar PekerjaPOST /workers akan set users.is_worker = true dan membuat baris baru di worker_profiles
Rating OtomatisRating di worker_profiles diupdate via trigger setiap kali ada INSERT/UPDATE di tabel reviews
Kategori Multi-selectSatu pekerja bisa memiliki banyak kategori — tersimpan di junction table worker_categories

Rekomendasi Stack Backend

LayerRekomendasiAlternatif
RuntimeNode.js (Express / Fastify)Go (Fiber), Python (FastAPI)
DatabasePostgreSQL 15+MySQL 8
ORMPrismaDrizzle, Knex
AuthJWT + bcryptAuth0, Supabase Auth
Storage GambarCloudinary / Supabase StorageAWS S3
Hosting DBSupabase / RailwayAWS RDS, Neon
Catatan Migrasi dari Frontend: Saat ini frontend menggunakan localStorage sebagai pengganti database (nyambi_user, nyambi_orders, nyambi_favorites, nyambi_reviews, nyambi_notifs). Saat backend sudah siap, ganti semua hook localStorage dengan pemanggilan ke API endpoint yang didokumentasikan di sini.
